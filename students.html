async function handleSaveRetroactive() { const retroMonth = document.getElementById('adminRetroMonth').value; const rawRooms = document.getElementById('adminRetroRooms').value.trim(); if (!rawRooms) { Swal.fire('แจ้งเตือน', 'กรุณาระบุห้องเรียนที่ต้องการเปิดระบบย้อนหลัง', 'warning'); return; } // แยกข้อความห้องเรียนด้วยคอมมา และตัดช่องว่างออก const roomsArray = rawRooms.split(',').map(r => r.trim()).filter(r => r.length > 0); Swal.fire({ title: 'กำลังบันทึกสิทธิ์...', allowOutsideClick: false, didOpen: () => { Swal.showLoading() } }); try { const res = await fetch(GAS_WEB_APP_URL, { method: 'POST', body: JSON.stringify({ action: "setRetroactiveAccess", teacherName: globalTeacherName, term: globalTerm, month: retroMonth, rooms: roomsArray }), headers: { 'Content-Type': 'text/plain;charset=utf-8' } }); const result = await res.json(); if (result.status === 'success') { Swal.fire({ icon: 'success', title: 'อนุมัติสิทธิ์สำเร็จ', text: `เปิดระบบเดือน${retroMonth} ให้ห้องเรียนที่เลือกเรียบร้อยแล้ว`, confirmButtonText: 'ตกลง' }).then(() => { document.getElementById('adminRetroRooms').value = ''; // เคลียร์ช่องกรอกข้อมูล }); } else { Swal.fire('ข้อผิดพลาด', result.message, 'error'); } } catch (e) { Swal.fire('Error', 'ไม่สามารถเชื่อมต่อฐานข้อมูลได้', 'error'); } }
